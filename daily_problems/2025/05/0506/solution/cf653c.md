**提示 1：** 注意当前序列是不满足要求的。

**提示 2：** 不满足要求的位置附近一定得换！

先看提示 1 和提示 2，为什么呢？如果不满足要求的位置附近都没有换过，那那个位置一定还是不满足要求啊！

所以，我们先找到一个不满足要求的位置，则我们就确定了交换的其中一个位置——接下来枚举 $n$ 种可能的另一个位置就行了。

那么如何检查某一种交换后是否能满足要求呢？

我们可以先预处理没有交换的情况下，有多少位置满足要求。而交换后，是否满足要求的改变只出现在交换的位置前后——我们只需检查这些位置是否从不满足要求变成满足要求 / 从满足要求变成不满足要求就行。最后检查满足要求的数量是否足够。

注意 $(u,u+1)$ 和 $(u+1,u)$ 是一种，这里的重复排除即可。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    nums = LII()
    
    def check(idx):
        return nums[idx] > nums[idx + 1] if idx & 1 else nums[idx] < nums[idx + 1]
    
    cur = [check(i) for i in range(n - 1)]
    res = sum(cur)
    
    vis = [0] * n
    
    ans = 0
    for i in range(n - 1):
        if not cur[i]:
            
            for u in range(i, i + 2):
                for v in range(n):
                    if u == i and v == i + 1:
                        continue
                    
                    nums[u], nums[v] = nums[v], nums[u]
                    nres = res
                    
                    for nu in range(u - 1, u + 1):
                        if 0 <= nu < n - 1 and not vis[nu]:
                            vis[nu] = 1
                            nres -= cur[nu]
                            nres += check(nu)
                    
                    for nv in range(v - 1, v + 1):
                        if 0 <= nv < n - 1 and not vis[nv]:
                            vis[nv] = 1
                            nres -= cur[nv]
                            nres += check(nv)
                    
                    if nres == n - 1:
                        ans += 1
                    
                    for nu in range(u - 1, u + 1):
                        if 0 <= nu < n - 1:
                            vis[nu] = 0
                    
                    for nv in range(v - 1, v + 1):
                        if 0 <= nv < n - 1:
                            vis[nv] = 0
                    
                    nums[u], nums[v] = nums[v], nums[u]
            
            break
    
    print(ans)
```

C++ 做法如下——

```cpp []
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    string s;

    vector<int> c1(26, 0), c2(26, 0);

    cin >> s;

    for (auto &c: s)
        c1[c - 'a'] ++;
    
    cin >> s;

    for (auto &c: s)
        c2[c - 'a'] ++;
    
    vector<int> v1, v2;
    
    for (int i = 0; i < 26; i ++)
        for (int j = 0; j < c1[i]; j ++)
            v1.emplace_back(i);
    
    for (int i = 25; i >= 0; i --)
            for (int j = 0; j < c2[i]; j ++)
                v2.emplace_back(i);
    
    int n = s.size();
    vector<int> ans(n, 0);

    int l = 0, r = n - 1, cnt1 = (n + 1) / 2, cnt2 = n / 2, p1 = 0, p2 = 0;

    for (int idx = 0; idx < n; idx ++) {
        if (idx % 2 == 0) {
            if (v1[p1] < v2[p2]) ans[l ++] = v1[p1 ++];
            else ans[r --] = v1[p1 + cnt1 - 1];
            cnt1 --;
        }
        else {
            if (v1[p1] < v2[p2]) ans[l ++] = v2[p2 ++];
            else ans[r --] = v2[p2 + cnt2 - 1];
            cnt2 --;
        }
    }

    for (auto &x: ans) cout << (char)(x + 'a');

    return 0;
}
```
