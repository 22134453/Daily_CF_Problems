**提示 1：** 

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    nums = LII()

    tmp = [[] for _ in range(n + 4)]
    for i in range(n):
        tmp[nums[i]].append(i)

    cur = 0
    ans = []

    for _ in range(n):
        while True:
            if tmp[cur]:
                ans.append(tmp[cur].pop())
                break
            else:
                if cur < 3: exit(print('Impossible'))
                cur -= 3
        cur += 1

    print('Possible')
    print(' '.join(str(x + 1) for x in ans))
```

C++ 做法如下——

```cpp []
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;

    vector<vector<int>> tmp(n + 4);
    for (int i = 0; i < n; i ++) {
        int x;
        cin >> x;
        tmp[x].emplace_back(i);
    }

    int cur = 0;
    vector<int> ans;

    for (int i = 0; i < n; i ++) {
        while (true) {
            if (tmp[cur].size()) {
                ans.emplace_back(tmp[cur].back());
                tmp[cur].pop_back();
                break;
            }
            else {
                if (cur < 3) return cout << "Impossible", 0;
                cur -= 3;
            }
        }
        cur ++;
    }

    cout << "Possible\n";
    for (auto &x: ans) cout << x + 1 << ' ';

    return 0;
}
```
